% ── Overview ─────────────────────────────────────────────────────────────────
\begin{frame}{Overview}
\begin{enumerate}
  \item Create a private GitHub repo (\texttt{a3-ci-test})
  \item Seed it with intentionally buggy Python code
  \item Run \texttt{a3 init .\ --copilot} to generate CI workflows
  \item Run \texttt{a3 scan} locally to preview findings
  \item Push to GitHub — CI triggers automatically
  \item Inspect CI results, SARIF, and Code Scanning dashboard
  \item Use \texttt{a3 baseline} ratchet for incremental adoption
\end{enumerate}
\end{frame}

% ── Step 6: CI Pipeline ─────────────────────────────────────────────────────
\begin{frame}{Step 6: What the CI Pipeline Does (5 stages)}
\begin{enumerate}
  \item \textbf{Checkout \& Setup} — \texttt{actions/checkout@v4}, Python 3.12, \texttt{pip install a3-python[ci]}
  \item \textbf{Get changed files} — \texttt{git diff} to find modified \texttt{*.py} files
  \item \textbf{Run a3 scan} — Static analysis $\to$ SARIF with barrier proofs + DSE
  \item \textbf{Agentic triage} — LLM agent explores repo (reads files, searches for guards, inspects callers/tests) using \texttt{GITHUB\_TOKEN} $\to$ classifies TP/FP
  \item \textbf{Baseline ratchet} — \texttt{a3 baseline diff}: \\
    \quad New bugs not in baseline $\Rightarrow$ \textcolor{red}{\textbf{CI fails}} \\
    \quad All bugs in baseline $\Rightarrow$ \textcolor{green}{\textbf{CI passes}}
  \item \textbf{Upload SARIF} — Results appear in Security $\to$ Code Scanning
\end{enumerate}
\end{frame}

% ── Step 7: Incremental Testing ─────────────────────────────────────────────
\begin{frame}[fragile]{Step 7: Incremental Testing with Baseline Ratchet}
\textbf{Accept current findings into baseline:}
\begin{lstlisting}[language=bash]
a3 baseline accept --sarif a3-results.sarif
git add .a3-baseline.json
git commit -m "ci: accept baseline findings"
git push
\end{lstlisting}

\textbf{Now CI passes.} Future pushes only fail on \emph{new} bugs.

\medskip
\textbf{Fix a bug and verify:}
\begin{lstlisting}[language=bash]
# Fix divide() to handle b==0
git commit -am "fix: guard divide() against zero"
git push
# CI runs -> no new bugs -> passes
# Baseline auto-prunes the fixed bug
\end{lstlisting}
\end{frame}

% ── Step 8: PR workflow ─────────────────────────────────────────────────────
\begin{frame}[fragile]{Step 8: Testing via Pull Request}
\begin{lstlisting}[language=bash]
git checkout -b add-new-function
# Add a new buggy function
echo 'def bad_func(x): return 1/x' >> src/calculator.py
git commit -am "feat: add new function"
git push -u origin add-new-function
gh pr create --title "Add new function" --body "Test a3 CI"
\end{lstlisting}

\textbf{What happens:}
\begin{itemize}
  \item CI triggers on the PR (only changed \texttt{*.py} files scanned)
  \item a3 finds new DIV\_ZERO in \texttt{bad\_func}
  \item Baseline diff: new bug not in baseline $\Rightarrow$ \textcolor{red}{\textbf{CI fails}}
  \item PR is blocked until bug is fixed or explicitly accepted
\end{itemize}
\end{frame}

\begin{frame}{Two Ways to Use a3 in CI}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Scenario A: Whole-Repo Scan}
\begin{itemize}
  \item Scan \emph{every} Python file at once
  \item Best for onboarding or scheduled audits
  \item Single command: \texttt{a3 scan .}
  \item Produces a full SARIF snapshot
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Scenario B: Incremental File-Add}
\begin{itemize}
  \item CI auto-invokes a3 whenever a \texttt{*.py} file is added or changed
  \item Best for day-to-day development
  \item Only changed files scanned
  \item Baseline ratchet blocks \emph{new} bugs only
\end{itemize}
\end{column}
\end{columns}

\bigskip
\centering
Both scenarios use the \textbf{same} workflow generated by \texttt{a3 init .\ --copilot}.
\end{frame}

% ── Scenario A ───────────────────────────────────────────────────────────────
\begin{frame}[fragile]{Scenario A: Whole-Repo Scan}
\textbf{Use case:} First-time adoption, weekly audit, or baseline generation.

\begin{lstlisting}[language=bash]
# Scan every Python file in the repository
a3 scan . --interprocedural --dse-verify --output-sarif results.sarif

# Optionally triage with agentic LLM
a3 triage --sarif results.sarif --provider github --agentic \
  --output-sarif triaged.sarif

# Accept current state as baseline (ratchet start)
a3 baseline accept --sarif triaged.sarif
git add .a3-baseline.json
git commit -m "ci: establish a3 baseline"
git push
\end{lstlisting}

\textbf{Result:}
\begin{itemize}
  \item Full picture of every reachable bug across the codebase
  \item Baseline locks in current findings — future CI only flags \emph{new} bugs
  \item Scheduled workflow (\texttt{a3-scheduled-scan.yml}) repeats this weekly
\end{itemize}
\end{frame}

% ── Scenario B ───────────────────────────────────────────────────────────────
\begin{frame}[fragile]{Scenario B: Incremental — Auto-Invoke on File Add}
\textbf{Use case:} A developer adds a new Python file. CI automatically invokes a3.

\begin{lstlisting}[language=bash]
# Developer adds a new file
cat > src/payments.py << 'EOF'
def charge(amount, discount):
    """Apply discount and charge."""
    return amount / discount   # DIV_ZERO: discount could be 0

def refund(amount, count):
    if count > 0:
        return amount / count  # SAFE: guarded
    return 0.0
EOF

git add src/payments.py
git commit -m "feat: add payment processing"
git push
\end{lstlisting}

\textbf{What happens automatically:}
\begin{enumerate}
  \item Push triggers \texttt{a3-pr-scan.yml} (path filter: \texttt{**.py})
  \item Workflow runs \texttt{git diff} $\to$ only \texttt{src/payments.py} scanned
  \item a3 finds DIV\_ZERO in \texttt{charge()}, proves \texttt{refund()} safe
  \item Baseline diff: new bug $\Rightarrow$ \textcolor{red}{\textbf{CI fails}} until fixed
\end{enumerate}
\end{frame}

% ── Scenario B: workflow trigger detail ──────────────────────────────────────
\begin{frame}[fragile]{Scenario B: How the Auto-Invoke Works}
The generated \texttt{a3-pr-scan.yml} contains:

\begin{lstlisting}[language=yaml]
on:
  push:
    branches: [main]
    paths: ["**.py"]          # <-- triggers ONLY when .py files change
  pull_request:
    branches: [main]
    paths: ["**.py"]
\end{lstlisting}

\textbf{Key behaviour:}
\begin{itemize}
  \item Adding \textbf{any} \texttt{.py} file $\to$ a3 auto-invokes (no manual step)
  \item Non-Python changes (docs, configs) $\to$ workflow does not run
  \item Changed files are detected via \texttt{git diff --name-only}
  \item Only changed files are scanned (fast incremental analysis)
\end{itemize}

\medskip
\textbf{No extra configuration needed.} This is the default after \texttt{a3 init .\ --copilot}.
\end{frame}

% ── Architecture ─────────────────────────────────────────────────────────────
\begin{frame}{Architecture: What Runs in CI}
\begin{center}
\begin{minipage}{0.85\textwidth}
\texttt{push/PR touches *.py} \\
\quad $\downarrow$ \\
\textbf{1. Static Analysis} (a3 scan) \\
\quad Bytecode analysis + Z3 symbolic execution \\

\quad $\downarrow$ (SARIF with surviving 1\%) \\
\textbf{2. Agentic LLM Triage} (a3 triage --agentic) \\
\quad LLM reads files, searches guards, checks callers \\
\quad Uses GITHUB\_TOKEN $\to$ GitHub Models (free) \\
\quad $\downarrow$ (filtered SARIF) \\
\textbf{3. Baseline Ratchet} (a3 baseline diff) \\
\quad New bugs $\Rightarrow$ CI fails \\
\quad $\downarrow$ \\
\textbf{4. Upload SARIF} (github/codeql-action/upload-sarif) \\
\quad Security $\to$ Code Scanning dashboard
\end{minipage}
\end{center}
